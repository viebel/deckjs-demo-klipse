function dijkstraAlgo(graph, id) {

 return graph.elements().dijkstra('#' + id, function(edge){
  return edge.data('weight');
});
}

function pathToArrows(path) {
  return path
    .filter((x,i) =&gt; i %2 === 0)
    .map(x =&gt; x.id()).join(" -&gt; ");
}
